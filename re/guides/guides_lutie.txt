xmas,140,137,3	script	Lutie Guide#xmas	717,{
	mes "[Lutie指南]";
	mes "歡迎來到 ^8b4513盧堤 ^000000，";
	mes "雪村。";
	mes "您需要幫助村莊嗎？";
	F_Navi("[Lutie指南]");
	while (1) {
		next;
		switch(select("[ 禮品商店 ]:[ 教堂 ]:[ 武器與防具商店 ]:[ 工具商店 ]:[ 聖誕老人之家 ]:[ 玩具工廠 ]:從迷你地圖移除標記:取消")) {
		case 1:
			callsub L_Mark, F_Navi("禮品商店","xmas,172,131","^008080");
			viewpoint 1,172,131,2,0x008080;
			break;
		case 2:
			callsub L_Mark, F_Navi("教堂","xmas,104,287","^006400");
			viewpoint 1,104,287,1,0xAAFF00;
			break;
		case 3:
			callsub L_Mark, F_Navi("武器與防具商店","xmas,171,158","^FF1493");
			viewpoint 1,171,158,3,0xFF1493;
			break;
		case 4:
			callsub L_Mark, F_Navi("工具商店","xmas,122,131","^8B4513");
			viewpoint 1,122,131,4,0x8B4513;
			break;
		case 5:
			callsub L_Mark, F_Navi("聖誕老人之家","xmas,149,237","^B9062F"),1;
			viewpoint 1,149,237,5,0x9400D3;
			break;
		case 6:
			callsub L_Mark, F_Navi("玩具工廠","xmas,143,312","^9400D3");
			viewpoint 1,143,312,6,0xFF0000;
			break;
		case 7:
			mes "[Lutie指南]";
			mes "當然，我將從您的迷你圖中刪除所有標記。";
			mes "我還有什麼可以為您做的嗎？";
			viewpoint 2,1,1,0,0xFFFFFF;
			viewpoint 2,1,1,1,0xFFFFFF;
			viewpoint 2,1,1,2,0xFFFFFF;
			viewpoint 2,1,1,3,0xFFFFFF;
			viewpoint 2,1,1,4,0xFFFFFF;
			viewpoint 2,1,1,5,0xFFFFFF;
			viewpoint 2,1,1,6,0xFFFFFF;
			break;
		case 8:
			mes "[Lutie指南]";
			mes "在旅途中安全。";
			close;
		}
	}
	end;
L_Mark:
	mes "[Lutie指南]";
	mes "讓我標記"+((getarg(1,0))?"":"這個");
	mes getarg(0)+"^000000";
	mes "在您的迷你圖上。";
	mes "您想檢查其他任何位置嗎？";
	return;
}
