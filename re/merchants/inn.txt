// Brasilis

bra_in01,27,24,3	script	Hotel Keeper#bra1	478,{
	mes "[酒店飼養員]";
	mes "歡迎來到美麗的巴西利斯酒店。";
	next;
	switch(select("保存:休息-5000 zeny")) {
	case 1:
		mes "[酒店飼養員]";
		mes "您想在巴西利斯酒店保存在這裡嗎？";
		next;
		switch(select("不，謝謝。 :當然。")) {
		case 1:
			mes "[酒店飼養員]";
			mes "好吧，享受您的住宿。";
			close;
		case 2:
			mes "[酒店飼養員]";
			mes "您的重生已在酒店保存下來。我希望您喜歡在巴西利斯逗留。";
			savepoint "bra_in01",144,69,1,1;
			close;
		}
	case 2:
		if (Zeny > 4999) {
			mes "[酒店飼養員]";
			mes "我會告訴你一個很棒的房間。";
			close2;
			set Zeny, Zeny-5000;
			percentheal 100,100;
			warp "bra_in01",144,69;
			end;
		}
		else {
			mes "[酒店飼養員]";
			mes "很抱歉，服務費是每晚5,000 Zeny。";
			close;
		}
	}
}
