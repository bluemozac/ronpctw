function	script	F_AuctionWarper	{
	mes "[拍賣廳指南]";
	mes "你好，你嗎？";
	mes "喜歡進入";
	mes "拍賣廳？";
	next;
	if (select("是:否") == 1) {
		.@num = getarg(0);
		if (.@num == 1 || .@num == 4) {
			mes "[拍賣廳指南]";
			mes "偉大的！好吧，那，";
			mes "我希望你玩得開心";
			mes "享受拍賣~";
		} else {
			mes "[拍賣廳指南]";
			mes "享受拍賣。";
		}
		close2;
		switch(.@num) {
		case 1:
			warp "auction_01",179,53; end;
		case 2:
			warp "auction_01",21,43; end;
		case 3:
			warp "auction_02",151,23; end;
		case 4:
			warp "auction_02",43,24; end;
		}
	}
	mes "[拍賣廳指南]";
	mes "好吧，那，";
	mes "稍後再見。";
	mes "如果你改變你的";
	mes "介意，請來";
	mes "享受拍賣~";
	close;
}
