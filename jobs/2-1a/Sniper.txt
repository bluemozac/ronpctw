valkyrie,44,55,5	script	Sniper#Valkyrie	727,{
	if (ADVJOB == 0 || Upper != 1) {
		set .@karma_d,rand(1,10);
		if (.@karma_d > 4) {
			mes "[狙擊手]";
			mes "恭喜。";
			mes "向勇士隊致敬！";
			close;
		}
		mes "[狙擊手]";
		mes "一槍。";
		mes "一個殺人。";
		mes "這不是那麼難";
		mes "一旦您開發了";
		mes "這種風格的願景";
		mes "戰鬥。";
		close;
	}
	else if (ADVJOB == Job_Sniper && Class == Job_Archer_High && JobLevel > 39) {
		mes "[狙擊手]";
		mes "世界在";
		mes "強大的弓箭手";
		mes "像你。你準備好了嗎";
		mes "很棒的責任？";
		next;
		mes "[狙擊手]";
		mes "你願意嗎";
		mes "邁出下一步";
		mes "成為狙擊手？";
		next;
		if (select("否:是") == 1) {
			mes "[狙擊手]";
			mes "準備好時，";
			mes "隨時回來。";
			mes "向勇士隊致敬！";
			close;
		}
		if (SkillPoint) {
			mes "[狙擊手]";
			mes "您仍然可以學習更多技能。請使用";
			mes "返回我之前，您剩下的所有技能點。";
			close;
		}
		jobchange Job_Sniper;
		set ADVJOB,0;
		mes "[狙擊手]";
		mes "恭喜！";
		mes "作為狙擊手，我希望";
		mes "那個邪惡的奴才";
		mes "永遠不會安全，所以";
		mes "只要他們在";
		mes "你的視線！";
		close;
	}
	mes "[狙擊手]";
	mes "歡迎";
	mes "到瓦爾哈拉，";
	mes "榮譽大廳。";
	next;
	mes "[狙擊手]";
	mes "請做";
	mes "自己舒適";
	mes "當你在這裡。";
	mes "向勇士隊致敬！";
	close;
}
