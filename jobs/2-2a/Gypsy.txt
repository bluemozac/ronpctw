valkyrie,53,56,3	script	Gypsy#Valkyrie	101,{
	if (ADVJOB == 0 || Upper != 1) {
		set .@karma_d,rand(1,10);
		if (.@karma_d > 4) {
			mes "[吉普賽人]";
			mes "恭喜。";
			mes "向勇士隊致敬！";
			close;
		}
		mes "[吉普賽人]";
		mes "向左移動，";
		mes "向右移動~！";
		mes "然後踩...！";
		mes "跳舞可以";
		mes "不僅僅是愛好。";
		mes "對我來說，這是一種生活方式~";
		close;
	}
	else if (ADVJOB == Job_Gypsy && Class == Job_Archer_High && JobLevel > 39) {
		mes "[吉普賽人]";
		mes "Midgard的土地";
		mes "需要才華橫溢的女性";
		mes "巧妙地改變餘額";
		mes "在良好之間的戰鬥中";
		mes "和邪惡。";
		next;
		mes "[吉普賽人]";
		mes "你準備好了嗎？";
		mes "為了扮演這個角色，";
		mes "成為吉普賽人？";
		next;
		if (select("否:是的。") == 1) {
			mes "[吉普賽人]";
			mes "準備好時，";
			mes "隨時回來。";
			mes "向勇士隊致敬！";
			close;
		}
		if (SkillPoint) {
			mes "[吉普賽人]";
			mes "您仍然可以學習更多技能。請使用";
			mes "返回我之前，您剩下的所有技能點。";
			close;
		}
		jobchange Job_Gypsy;
		set ADVJOB,0;
		mes "[吉普賽人]";
		mes "恭喜！";
		mes "作為吉普賽人，我知道";
		mes "你的表演";
		mes "搖擺所有人的心";
		mes "那些會看的人...";
		close;
	}
	mes "[吉普賽人]";
	mes "歡迎";
	mes "到瓦爾哈拉，";
	mes "榮譽大廳。";
	next;
	mes "[吉普賽人]";
	mes "請做";
	mes "自己舒適";
	mes "當你在這裡。";
	mes "向勇士隊致敬！";
	close;
}
