valkyrie,53,39,3	script	Paladin#Valkyrie	752,{
	if (ADVJOB == 0 || Upper != 1) {
		set .@karma_d,rand(1,10);
		if (.@karma_d > 4) {
			mes "[聖騎士]";
			mes "恭喜。";
			mes "向勇士隊致敬！";
			close;
		}
		mes "[聖騎士]";
		mes "不要忘記";
		mes "那個聖戰";
		mes "快到了！";
		mes "我們必須準備好自己！";
		mes "願正義之光";
		mes "總是照亮我們的道路！";
		close;
	}
	else if (ADVJOB == Job_Paladin && Class == Job_Swordman_High && JobLevel > 39) {
		mes "[聖騎士]";
		mes "聖戰將";
		mes "在我們之前待在我們之前";
		mes "知道。比以往任何時候都多";
		mes "我們需要堅強的男人";
		mes "和女人為什麼而戰";
		mes "很好，正確。";
		next;
		mes "[聖騎士]";
		mes "你會打架嗎";
		mes "公義的一面";
		mes "作為聖騎士？";
		next;
		if (select("否:是的。") == 1) {
			mes "[聖騎士]";
			mes "準備好時，";
			mes "隨時回來。";
			mes "向勇士隊致敬！";
			close;
		}
		if (SkillPoint) {
			mes "[聖騎士]";
			mes "您仍然可以學習更多技能。請使用";
			mes "返回我之前，您剩下的所有技能點。";
			close;
		}
		jobchange Job_Paladin;
		set ADVJOB,0;
		mes "[聖騎士]";
		mes "恭喜。";
		mes "作為聖騎士，我希望";
		mes "你會保護那些";
		mes "比你弱，帶來";
		mes "美國在即將到來的";
		mes "善與惡之間的戰爭。";
		close;
	}
	mes "[聖騎士]";
	mes "歡迎";
	mes "到瓦爾哈拉，";
	mes "榮譽大廳。";
	next;
	mes "[聖騎士]";
	mes "請做";
	mes "自己舒適";
	mes "當你在這裡。";
	mes "向勇士隊致敬！";
	close;
}
