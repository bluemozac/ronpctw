prontera,128,200,6	script	Platinum Skill NPC	94,{
	mes "[鉑技能NPC]";
	mes "我可以為您提供您工作的特殊技能。";
	mes "您現在想要這些技能嗎？";
	next;
	if (select("是的，請:否") == 2) {
		mes "[鉑技能NPC]";
		mes "祝您有美好的一天...>。 >";
		close;
	}
	callfunc "F_GetPlatinumSkills";
	mes "[鉑技能NPC]";
	mes "你去！";
	close;
}
