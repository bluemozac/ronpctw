prontera,113,294,3	script	Sign Post#god	837,{
	mes "=========上帝印章狀態==========";
	setarray .@seal$[1], "Sleipnir","Megingjard","Brisingamen","Mjolnir";
	for(set .@i,1; .@i<=4; set .@i,.@i+1) {
		set .@val, getd("$God"+.@i);
		if (.@val == 0)
			set .@status$,"Unseen";
		else if (.@val < $@god_check1)
			set .@status$,"Active";
		else if (.@val < $@god_check2)
			set .@status$,"Appeared";
		else
			set .@status$,"Released";
		mes .@seal$[.@i]+"海豹："+.@status$;
	}
	mes " ";
	mes "=======您的印章狀態===========";
	if (god_sl_1 < 51)
		mes "^ff0000sleipnir密封：未完成^000000";
	else
		mes "^00ff00sleipnir密封：完成^000000";
	if (god_eremes < 28)
		mes "^ff0000megingjard密封：未完成^000000";
	else
		mes "^00ff00megingjard密封：完成^000000";
	if (god_brising < 50)
		mes "^FF0000 Brisingamen密封：未完成^000000";
	else
		mes "^00ff00 brisingamen密封：完成^000000";
	if (god_mjo_0 < 11)
		mes "^FF0000MJOLNIR密封：未完成^000000";
	else
		mes "^00FF00MJOLNIR密封：完成^000000";
	close;
}
